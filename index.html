library(tidyverse)
library(ggplot2)
library(readr)
library(plotly)
library(DT)
library(dplyr)
#Carga del archivo CSV de entrada en un dataframe con la funci√≥n
#read_delim() de readr
covid_general <-
  read_delim(
    file = "c:/r/quarto/Datoscovid/05_30_22_CSV_GENERAL.csv",
    col_select = c(
      "FECHA",
      "hom_posi",
      "muj_posi",
      "menor_posi",
      "adul_posi",
      "am_posi",
      "nue_posi"
    )
   )

# Cambio de nombre de columnas
covid_general <-
  covid_general |>
  rename(
    Hombres_positivos = hom_posi,
    Mujeres_positivos = muj_posi,
    Menores_positivos = menor_posi,
    Adultos_positivos = adul_posi,
    Mayores_positivos = am_posi,
    Nuevos_positivos =  nue_posi
  )

# Cambio de tipo de datos de la columna fecha, de str a date
covid_general <-
  covid_general |>
  mutate(FECHA = as.Date(FECHA, format = "%d/%m/%Y"))
# Tabla de datos de COVID generales
covid_general |>
  datatable(options = list(
    pageLength = 5,
    language = list(url = '//cdn.datatables.net/plug-ins/1.10.11/i18n/Spanish.json')
  ))
# Gr√°fico de barras Cantidad de casos positivos nuevos por d√≠a
grafico_barras <-
covid_general |>
  ggplot(aes(x = FECHA, y = Nuevos_positivos)) +
  geom_col() +
  ggtitle("Cantidad de casos positivos nuevos por d√≠a") +
  xlab("fecha") +
  ylab("Cantidad casos nuevos positivos") 

# Gr√°fico de barras plotly
ggplotly(grafico_barras) |> 
  config(locale = "es")
# Gr√°fico de l√≠neas evoluci√≥n de los casos positivos acumulados de COVID por g√©nero
grafico_lineas_ggplot2 <-
  covid_general |>
  ggplot(aes(x = FECHA, y = value, color = variable)) +
  geom_line(aes(y = Hombres_positivos, color = "Hombres")) +
  geom_line(aes(y = Mujeres_positivos, color = "Mujeres")) +
  scale_color_manual( # colores
    "",
    values = c(
      "Hombres" = "blue",
      "Mujeres" = "green"
    )
  ) +
  ggtitle("Evoluci√≥n de los Casos Positivos acumulados por g√©nero") +
  xlab("Fecha") +
  ylab("Casos") 
    

# Gr√°fico de dispersi√≥n plotly
ggplotly(grafico_lineas_ggplot2) |>
  config(locale = 'es')
# Gr√°fico de l√≠neas con la evoluci√≥n de los casos positivos acumulados de COVID por grupo de edad
grafico_lineas_ggplot2 <-
  covid_general |>
  ggplot(aes(x = FECHA, y = value, color = variable)) +
  geom_line(aes(y = Menores_positivos, color = "Menores")) +
  geom_line(aes(y = Adultos_positivos, color = "Adultos")) +
  geom_line(aes(y = Mayores_positivos, color = "Adultos Mayores")) +
  scale_color_manual( # colores
    "",
    values = c(
      "Menores" = "blue",
      "Adultos" = "green",
      "Adultos Mayores" = "red"
    )
  ) +
  ggtitle("Evoluci√≥n de los Casos Positivos acumulados por grupo de edad") +
  xlab("Fecha") +
  ylab("Casos positivos grupo etario") 
    

# Gr√°fico de l√≠neas plotly
ggplotly(grafico_lineas_ggplot2) |>
  config(locale = "es")
# Carga de CSV_POSITIVOS y lectura de datos de inter√©s
covid_cantonal_positivos <-
  read_delim(
    file = "c:/r/quarto/Datoscovid/05_30_22_CSV_POSITIVOS.csv",
    locale = locale(encoding = "WINDOWS-1252"),
    col_select = c("provincia", "canton", "30/05/2022")
  )

# Cambio de nombre de columnas
covid_cantonal_positivos <-
  covid_cantonal_positivos |>
  rename(
    positivos_al_30_mayo_22 = '30/05/2022'
  )


# Borrado de filas con valor NA u "Otros"
# en la columna canton
covid_cantonal_positivos <-
  covid_cantonal_positivos |>
   filter(!is.na(canton) & canton != "Otros")

# Tabla de datos de COVID cantonales positivos
covid_cantonal_positivos |>
  datatable(options = list(
    pageLength = 5,
    language = list(url = '//cdn.datatables.net/plug-ins/1.10.11/i18n/Spanish.json')
  ))
#Histograma Casos Positivos en cantones
histograma_ggplot2 <-
covid_cantonal_positivos |>
  ggplot(aes(x = positivos_al_30_mayo_22)) +
  geom_histogram(bins = 5) +
  ggtitle("Distribuci√≥n de Casos Positivos en Cantones al 30 de mayo de 2022") +
  xlab("Casos") +
  ylab("Frecuencia") +
  labs(subtitle = "Datos")

# Histograma plotly
ggplotly(histograma_ggplot2) |> 
  config(locale = 'es')
